    function uploadImage() {
      const fileInput = document.getElementById("imageDisplayUrl");
      const file = fileInput.files[0];

      if (file) {
        // Image file is selected
        const apiKey = "ba238be3f3764905b1bba03fc7a22e28";
        const uploadUrl = "https://api.imgbb.com/1/upload";

        const formData = new FormData();
        formData.append("key", apiKey);
        formData.append("image", file);

        fetch(uploadUrl, {
          method: "POST",
          body: formData
        })
          .then(response => response.json())
          .then(data => {
            if (data.success) {
              // Image uploaded successfully
              const imageUrl = data.data.medium.url; // Extract the medium display URL
              generateJSON(imageUrl);
            } else {
              // Image upload failed
              console.error("Image upload failed:", data.error.message);
              generateJSON();
            }
          })
          .catch(error => {
            console.error("Error uploading image:", error);
            generateJSON();
          });
      } else {
        // Image file is not selected
        generateJSON();
      }
    }
